  <script>
    // Store selected answers
    let userAnswers = {
      time: "",
      restaurant: "",
      nightcap: ""
    };

    // Utility function to show/hide steps
    function goToStep(stepId) {
      // Hide all steps
      document.querySelectorAll('.step').forEach(step => {
        step.classList.remove('active');
      });

      // If user clicked "I don't like you"
      if (stepId === 'sad') {
        document.getElementById('sadPage').classList.add('active');
        return;
      }

      // Show the intended step
      if (stepId === 2) {
        document.getElementById('step2').classList.add('active');
      } else if (stepId === 3) {
        // Save time from step 2
        const timeVal = document.getElementById('timeInput').value;
        userAnswers.time = timeVal;
        document.getElementById('step3').classList.add('active');
      } else if (stepId === 4) {
        document.getElementById('step4').classList.add('active');
      } else if (stepId === 5) {
        document.getElementById('finalStep').classList.add('active');
        // Fill in final answers
        document.getElementById('finalTime').innerText = userAnswers.time || "(No time entered)";
        document.getElementById('finalRestaurant').innerText = userAnswers.restaurant || "(No selection)";
        document.getElementById('finalNightcap').innerText = userAnswers.nightcap || "(No selection)";
      } else {
        // Step 1 by default
        document.getElementById('step1').classList.add('active');
      }
    }

    // Choice selection (restaurant, nightcap)
    function selectChoice(type, value, element) {
      // Clear existing selection
      let containerId = (type === 'restaurant') ? 'restaurantChoices' : 'nightcapChoices';
      document.getElementById(containerId).querySelectorAll('.choice-item').forEach(item => {
        item.classList.remove('selected');
      });
      // Mark this as selected
      element.classList.add('selected');
      // Save user answer
      userAnswers[type] = value;
    }
  </script>
